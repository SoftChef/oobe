(static) deepObjectAssign(target, sources) → {object}

Source:
  • Helper.js, line 174

Object concat.

Example
let target = {
    a: 5,
    b: 10,
    c: {
        a: 7,
        b: 8
    }
}
let output = deepObjectAssign(target, {
    a: 8,
    c: {
        a: 10
    }
})
console.log(output.a) // 8
console.log(output.b) // 10
console.log(output.c.a) // 10
console.log(output.c.b) // 8
